11.00 — SETTINGS SYSTEM
  11.00.001 — Create src/settings/Settings.hpp/cpp
    - struct Settings { DisplaySettings display; AudioSettings audio; InputSettings input; AccessibilitySettings access; GameplaySettings gameplay; };
    - struct DisplaySettings { int cell_width=16; int cell_height=16; bool truecolor=true; int font_size=100; std::string palette="default"; float ui_scale=1.0; bool vsync=true; bool show_fps=false; bool show_debug=false; };
    - struct AudioSettings { float master_volume=1.0; float sfx_volume=1.0; float music_volume=0.7; float ambient_volume=0.5; bool mute=false; };
    - struct InputSettings { int keybind_profile=0; bool smooth_camera=true; float camera_lerp=0.2; float mouse_sensitivity=1.0; bool pause_on_focus_loss=true; };
    - struct AccessibilitySettings { int text_size=100; bool high_contrast=false; bool colorblind_mode="none"; bool reduce_motion=false; bool screen_reader_hints=false; float ui_scale=1.0; };
    - struct GameplaySettings { int difficulty=1; bool permadeath=false; bool auto_save=true; int auto_save_minutes=10; bool show_damage_numbers=false; bool pause_during_dialogue=true; float timescale_default=1.0; };
  11.00.002 — Load from XDG_CONFIG_HOME/ash/settings.json on startup
  11.00.003 — Save on change (debounced 1s)
  11.00.004 — Hot-reload settings without restart where possible
  11.00.005 — Validate schema on load, reset invalid fields to defaults
  11.00.006 — Settings UI (menu) with categorized panels

11.12 — PERFORMANCE BASELINE
  11.12.001 — Target: 60 FPS at 1920x1080 on mid-range hardware (Intel UHD 620)
  11.12.002 — 200 visible entities without frame drops
  11.12.003 — 80 maps loaded simultaneously in memory (~80MB)
  11.12.004 — Save game <2s, load game <4s
  11.12.005 — Startup time <3s on SSD
  11.12.006 — Memory footprint <500MB resident
  11.12.007 — Binary size <50MB (release)
  11.12.008 — Update budget per frame: 16ms (60 FPS), budget per system:
    - Input: 0.5ms
    - AI: 3ms
    - Combat: 2ms
    - Render: 4ms
    - Quest: 1ms
    - UI: 1ms
    - Save (background): async
  11.12.009 — Profile each system, surface slow frames in debug overlay
  11.12.010 — Add 30 micro-benchmarks using Google Benchmark or Catch2 BENCHMARK macros
